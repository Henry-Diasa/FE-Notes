工厂模式其实就是将创建对象的过程单独封装。通过一种类型来判断使用不同的构造函数

``` 
    function User(name,age,career,work){
        this.name = name
        this.age = age
        this.career = career
        this.work = work
    }

    // 工厂

    function factory(name,age,career){
        let work
        switch(career){
            case 'coder':
                work = ['写代码','写系分','修bug']
                break
            case 'product manager':
                work = ['定会议室','写PRD','催更'] 
                break
            case 'boss':
                work = ['喝茶','看报','见客户']
            case 'xxx'           
        }

        return new User(name,age,career,work)
    }
```
